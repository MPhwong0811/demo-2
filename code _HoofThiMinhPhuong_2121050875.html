<script>
let cartItems = [];
window.onload = function () {
displayProducts(products);
};
function search(){
let searchTerm = document.querySelector('#search input').value.toLowerCase();
let filteredProducts = products.filter(product => product.name.toLowerCase().includes(searchTerm));
displayProducts(filteredProducts);
}
function addToCart(productName, price) {
let quantityElement=document.getElementById('quantity-${productName}')
let quantity = parseInt(document.getElementById('quantity').value);
let total = quantity * price;
cartItems.push({ productName, quantity, total });
displayCart()
}
function displayCart(){
let cartList = document.getElementById('cart-items');
cartList.innerHTML = '';
let total = 0;
for (let item of cartItems) {
total += item.total;
let listItem = document.createElement('li');
listItem.textContent = `${item.quantity} x ${item.productName} - ₫${formatCurrency(item.total)}`;
cartList.appendChild(listItem);
}
document.getElementById('total').textContent = `Tổng tiền: ₫${formatCurrency(total)}`;
}
function formatCurrency(amount) {
return new Intl.NumberFormat('vi-VN', { style: 'currency', currency: 'VND' }).format(amount);
}
